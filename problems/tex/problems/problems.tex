\begin{problem}
$\forall L \in \mathtt{NatList} \colon \rmx{fold*}(\rmx{rev}(L)) = \rmx{fold*}(L)$.
\end{problem}

\end{problem}

\end{problem}

%%% Problem 4

\end{problem}

%%% Problem 5

\end{problem}
